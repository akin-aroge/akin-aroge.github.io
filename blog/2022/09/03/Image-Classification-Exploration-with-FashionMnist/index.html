class Scaling(tf.keras.layers.Layer):
    def __init__(self, scale, **kwargs):
        super(Scaling, self).__init__(**kwargs)
        self.scale = scale
        self.trainable = False
        #self.compute_dtype = tf.float32  # read_only
        
    def call(self, inputs):
        dtype = self.compute_dtype
        inputs = tf.cast(inputs, tf.float32)
        scale = tf.cast(self.scale, tf.float32)
        return  inputs * scale
    
    def get_config(self):
        config = super(Scaling, self).get_config()
        config.update({'scale':self.scale})
        return config

# layer insertion method into sequetial model
def insert_layer_untrained(model, loc_idx, new_layer):
    
    
    old_layers = model.layers.copy()
    old_layers.insert(loc_idx, new_layer)
    
    new_model = tf.keras.Sequential(old_layers)
    
    return new_model

CNN model with data augmentation¶

Given the pixelated nature of the images, an erosion may aid to further discern actual boundary pixel from just noise pixels. We can also increase the iniput data variety by performing a random horizontal flip on the data during training.

First we create and RandomErosion class by subclassing the Layer class, whereby erosion is randomly applied to the input image (batches) during training, with a probability of 0.2 for each batch. The keras RandomFlip layer is used for horizontal flipping

class RandomErosion(tf.keras.layers.Layer):
    def __init__(self, filter_=tf.zeros((3,3,1)), prob=0.2, name=None, **kwargs) -> None:
        super(RandomErosion, self).__init__(name=name, **kwargs)
        #self.input_spec = tf.keras.layers.InputSpec(ndim=4)
        self.g = tf.random.Generator.from_seed(1234)
        self.filter_ = filter_
        self.prob = prob
        self.trainable=False
    
    def call(self, inputs, training=None):
        if training:
            #tf.print('training')
            inputs = tf.cast(inputs, tf.float32)
            #bernoulli trial
            val = tf.cast(self.g.uniform(shape=[1], minval=0, maxval=1) < tf.constant(self.prob), tf.int32)[0]
            def true_fn():
                # random int between 0 and 3
                #tf.print('erosion')
                filter_ = tf.cast(self.filter_, tf.float32)
                rot = tf.random.uniform((),0,4, dtype=tf.int32)
                output = tf.nn.erosion2d(inputs, filters=filter_, 
                             strides=(1,1,1,1),
                            padding='SAME',
                            data_format='NHWC',
                            dilations=[1, 1, 1, 1])
                return output
            
            def false_fn():
                return inputs

            output = tf.cond(tf.equal(val, 1), true_fn=true_fn, false_fn=false_fn)
        else:
            #tf.print('no training')
            output = inputs
        
        return output
    
    def compute_output_shape(self, input_shape):
        return input_shape

def plot_confusion_matrix(y_test, y_pred, class_names, ax=None):
    
    if ax is None:
        ax = plt.gca()
        
    fig = plt.gcf()
    fig.set_size_inches(8,8)
    
    from sklearn.metrics import confusion_matrix
    mat = confusion_matrix(y_test, y_pred,   normalize="true")
    
    import seaborn as sns
    sns.heatmap(mat, square=True, annot=True, cbar=False)
    
    all_labels = np.arange(len(class_names)) + 0.5  # to shift to middle
    ax.set_xticks(all_labels, class_names, rotation=90, )
    ax.set_yticks(all_labels, class_names, rotation=0, )
    ax.set_xlabel('predicted value')
    ax.set_ylabel('true value');
    
    
def plot_image(idx, im, pred_array, true_label, class_names, ax=None):
    
    if ax is None:
        ax = plt.gca()

    pred_class = class_names[pred_label]
    ax.set_title(pred_class, color = 'green' if (pred_label==true_label) else 'red')
    ax.axis('off')
    return ax

def  get_trained_model(path, model, x_train, y_train, x_test, y_test, retrain=False, epoch=10, **kwargs):
    
    model_path = path
    epoch = kwargs.get('epoch', epoch)
    
 
    from keras.callbacks import CSVLogger
    csv_logger = CSVLogger(path+'/training.log', separator=',', append=False)
    
    if retrain is False:
        try:
            model =tf.keras.models.load_model(path)
        except OSError:
            model.fit(x=x_train, y=y_train, epochs=epoch, validation_data=(x_test, y_test), callbacks=[csv_logger], kwargs )
            model.save(model_path)
    else:
        model.fit(x=x_train, y=y_train, epochs=epoch, validation_data=(x_test, y_test), callbacks=[csv_logger]  )
        model.save(model_path)  
    
    import pandas as pd
    log_data = pd.read_csv(path+'/training.log', sep=',', engine='python')
     
    return model, log_data

        self.true_fn = true_fn(inputs)
        self.false_fn = false_fn(inputs)
        
    def true_fn(inputs):
        print('eroding')
        inputs = tf.cast(inputs, tf.float32)
        filter = tf.cast(self.filter, tf.float32)
        output = tf.nn.erosion2d(inputs, filters=filter, 
                                 strides=(1,1,1,1),
                                padding='SAME',
                                data_format='NHWC',
                                dilations=[1, 1, 1, 1])
        return output

    def false_fn(inputs):
        print('no eroding')
        return inputs
    
    def call(self, inputs):
        val = tf.cast(g.uniform(shape=[1], minval=0, maxval=1) < tf.constant(self.prob), tf.int32)[0]
        


        
        output = tf.cond(val==1, 
                        true_fn= true_fn, false_fn=false_fn)
        
        return output

        config = super(RandomErosion, self).get_config()
